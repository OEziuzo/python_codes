#Number Guessing Game:
from art import logo
import random

#Welcome message with ASCII art logo.
print(logo)
print("Welcome to the Number Guessing Game!\nI'm thinking of a number between 1 and 100")

#Difficulty level from the user.
difficulty = input("Choose a difficulty. Type 'easy' or 'hard': ").strip().lower()
if difficulty == "easy":
  lifeline = 10
  print(f"You have {lifeline} attempts remaining to guess the number")
elif difficulty == "hard":
  lifeline = 5
  print(f"You have {lifeline} attempts remaining to guess the number")

#Chosing a random integer between 1 and 100.
random_number = random.randint(1, 100)
print(random_number)
guess = int(input("Make a guess: "))

#Function to compare user's guess with random number.
def compare():
  if guess > random_number:
    print("Too high")
  elif guess < random_number:
    print("Too low")
  else:
    print(f"You got it! The answer was {random_number}")
  
compare()
#Repetition of the process untill the conditions are met
while lifeline != 1:
  if guess != random_number:
    guess = int(input("Guess again "))
    compare()
    lifeline -= 1
  
print("You've currently run out of guesses, you lose.")
